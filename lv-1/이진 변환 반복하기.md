```python
def solution(s):
    zero = tran = 0
    while True:
        if s == '1':
            break
        tran += 1
        zero += s.count('0')
        s = ''.join(s.split('0'))
        num, arr = len(s), []
        while num > 0:
            arr.append(num % 2)
            num //= 2
        s = ''.join(map(str, reversed(arr)))
    answer = [tran, zero]
    return answer

```
