```python
def solution(s):
    answer = ''
    words = s.split(' ')
    arr = []
    for word in words:
        tmp = ''
        for i, ch in enumerate(word):
            if i % 2 == 0:
                tmp += ch.upper()
            else:
                tmp += ch.lower()
        arr.append(tmp)
    answer = ' '.join(arr)
    return answer

```
