## CODE 1

```javascript
let pair = { ")": "(", "]": "[" };
for (let sen of input) {
  if (sen === ".") break;
  let stk = [];
  let ans = "yes";
  for (let ch of sen) {
    if ("([".includes(ch)) {
      stk.push(ch);
      continue;
    }
    if (!")]".includes(ch)) continue;
    if (stk.at(-1) === pair[ch]) stk.pop();
    else {
      ans = "no";
      break;
    }
  }
  if (stk.length > 0) ans = "no";
  console.log(ans);
}
```
