```python
def solution(s, n):
    answer = ''
    for ch in s:
        if not ch.isalpha():
            answer += ch
            continue
        tmp = ord(ch) + n
        if ch.isupper():
            if tmp > ord('Z'):
                tmp += ord('A') - ord('Z') - 1
        else:
            if tmp > ord('z'):
                tmp += ord('a') - ord('z') - 1
        answer += chr(tmp)
    return answer

```
