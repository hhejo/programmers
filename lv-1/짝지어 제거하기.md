```python
def solution(s):
    answer = -1
    while True:
        if len(s) == 0:
            answer = 1
            break
        stk = []
        for ch in s:
            if stk and stk[-1] == ch:
                stk.pop()
            else:
                stk.append(ch)
        tmp = ''.join(stk)
        if s == tmp:
            answer = 0
            break
        s = tmp
    return answer

```
