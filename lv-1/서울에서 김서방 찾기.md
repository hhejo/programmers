## Solution 2

Code

```javascript
function solution(seoul) {
  let target = 0;
  for (let [idx, val] of seoul.entries()) {
    if (val === 'Kim') {
      target = idx;
      break;
    }
  }
  let ans = `김서방은 ${target}에 있다`;
  return ans;
}
```
