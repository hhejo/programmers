## CODE 1

```python
def solution(s):
    s = s[2:-2]
    answer = []
    arr = s.split('},{')
    arr.sort(key=len)
    for x in arr:
        for num in map(int, x.split(',')):
            if num not in answer:
                answer.append(num)
    return answer

```
