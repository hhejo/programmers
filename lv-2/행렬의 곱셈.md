## CODE 1

```javascript
function solution(arr1, arr2) {
  let answer = [];
  for (let i = 0; i < arr1.length; i++) {
    let arr = [];
    for (let j = 0; j < arr2[0].length; j++) {
      let acc = 0;
      for (let k = 0; k < arr2.length; k++) {
        acc += arr1[i][k] * arr2[k][j];
      }
      arr.push(acc);
    }
    answer.push(arr);
  }
  return answer;
}
```
