## 풀이 1

```javascript
function solution(m, n, board) {
  const g = board.map((el) => [...el]);
  let ans = 0;
  while (true) {
    const v = Array.from(Array(m), () => new Array(n).fill(0));
    // 2x2 블록 확인
    for (let x = 0; x < m - 1; x++) {
      for (let y = 0; y < n - 1; y++) {
        if (g[x][y] === '') continue;
        else if (g[x][y] !== g[x][y + 1]) continue;
        else if (g[x][y + 1] !== g[x + 1][y]) continue;
        else if (g[x + 1][y] !== g[x + 1][y + 1]) continue;
        v[x][y] = 1;
        v[x][y + 1] = 1;
        v[x + 1][y] = 1;
        v[x + 1][y + 1] = 1;
      }
    }
    // 사라질 블록 지우고 개수 세기
    let cnt = 0;
    for (let x = 0; x < m; x++) {
      for (let y = 0; y < n; y++) {
        if (v[x][y] === 0) continue;
        cnt += 1;
        g[x][y] = '';
      }
    }
    // 지워진 블록이 없으면 종료
    if (cnt === 0) break;
    ans += cnt;
    // 블록 내리기
    for (let y = 0; y < n; y++) {
      let idx = m - 1;
      while (true) {
        let space = false;
        for (let x = idx; x >= 0; x--) {
          if (g[x][y] === '') {
            idx = x;
            space = true;
            break;
          }
        }
        if (!space) break;
        let char = false;
        for (let x = idx - 1; x >= 0; x--) {
          if (g[x][y] === '') continue;
          [g[idx][y], g[x][y]] = [g[x][y], g[idx][y]];
          idx -= 1;
          char = true;
        }
        if (!char) break;
      }
    }
  }
  return ans;
}
```

```javascript
function solution(m, n, board) {
  let b = board.map((el) => [...el]);
  let b2 = board.map((el) => [...el]);
  while (true) {
    let cnt = 0;
    for (let x = 0; x < m - 1; x++) {
      for (let y = 0; y < n - 1; y++) {
        if (b[x][y] === '') continue;
        else if (b[x][y] !== b[x][y + 1]) continue;
        else if (b[x][y + 1] !== b[x + 1][y]) continue;
        else if (b[x + 1][y] !== b[x + 1][y + 1]) continue;
        [b2[x][y], b2[x][y + 1]] = ['', ''];
        [b2[x + 1][y], b2[x + 1][y + 1]] = ['', ''];
        cnt += 1;
      }
    }
    if (cnt === 0) break;

    for (let x = 0; x < m - 1; x++) {
      for (let y = 0; y < n; y++) {
        if (b2[x + 1][y] !== '') continue;
        for (let i = x; i >= 0; i--) {
          b2[i + 1][y] = b2[i][y];
          b2[i][y] = '';
        }
      }
    }
    b = b2.map((el) => [...el]);
  }

  const blocks = b2.reduce((acc, cur) => acc.concat(cur));
  const ans = blocks.filter((el) => el === '').length;
  return ans;
}
```
