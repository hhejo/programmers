## CODE 1

```javascript
function solution(n) {
  let ans = 0;
  while (n) {
    if (n % 2 === 0) {
      n /= 2;
      continue;
    }
    n -= 1;
    ans += 1;
  }
  return ans;
}
```

## CODE 2

```javascript
function solution(n) {
  let count = 0;
  while (true) {
    if (n === 1) break;
    if (n % 2 === 0) n /= 2;
    else {
      n--;
      count++;
    }
  }
  count++;
  return count;
}
```
