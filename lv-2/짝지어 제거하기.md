## CODE 1

```javascript
function solution(s) {
  let stack = [];
  for (let i = 0; i < s.length; i++) {
    let curCh = s.at(i);
    if (stack.at(-1) === curCh) stack.pop();
    else stack.push(curCh);
  }
  return stack.length === 0 ? 1 : 0;
}
```
