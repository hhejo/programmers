## 풀이 1

```javascript
function solution(msg) {
  let map = new Map();
  for (let i = 0; i < 26; i++) map.set(String.fromCharCode(65 + i), i + 1);
  let [lastWordNum, wIdx, answer] = [27, 0, []];
  while (wIdx < msg.length) {
    let [word, num, cIdx] = [msg[wIdx], 0, wIdx + 1];
    while (map.has(word)) {
      num = map.get(word);
      word += msg[cIdx++];
    }
    map.set(word, lastWordNum++);
    wIdx = cIdx - 1;
    answer.push(num);
  }
  return answer;
}
```
