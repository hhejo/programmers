## CODE 1

```javascript
function solution(begin, end) {
  function getMaxDiv(num) {
    if (num === 1) return 0;
    const arr = [];
    for (let i = 2; i * i <= num; i++) {
      if (num % i === 0) {
        arr.push(i);
        if (num / i <= 1e7) return num / i;
      }
    }
    if (arr.length >= 1) return arr.at(-1);
    return 1;
  }
  const answer = [];
  for (let i = begin; i <= end; i++) answer.push(getMaxDiv(i));
  return answer;
}
```
