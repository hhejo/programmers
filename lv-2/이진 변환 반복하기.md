## CODE 1

```javascript
function solution(s) {
  let [cnt, z] = [0, 0];
  while (s !== "1") {
    cnt += 1;
    z += [...s].filter((x) => x === "0").length;
    s = s.split("0").join("");
    let [num, arr] = [s.length, []];
    while (num > 0) {
      arr.push(num % 2);
      num = Math.floor(num / 2);
    }
    s = arr.reverse().join("");
  }
  return [cnt, z];
}
```
