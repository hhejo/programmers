## CODE 1

function solution(n) {
  let ans = 0;
  for (let x = 1; x <= n; x++) {
    let sum = 0;
    for (let y = x; y <= n; y++) {
      sum += y;
      if (sum >= n) {
        if (sum === n) ans += 1;
        break;
      }
    }
  }
  return ans;
}
```

function solution(n) {
  let [left, right] = [1, 2];
  let [total, count] = [1, 0];
  while (left < right) {
    if (total < n) {
      total += right++;
    } else if (total > n) {
      total -= left++;
    } else {
      count++;
      total -= left++;
      total += right++;
    }
  }
  return count;
}
```
